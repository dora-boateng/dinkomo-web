131     {
132         // Retrieve the language object.
133         if (!$lang = Language::find($id)) {
134             abort(404);
135         }
136 
137         $this->validate($this->request, (new Language)->validationRules);
138 
139         // Update attributes.
140         $lang->fill($this->request->only([
141             'parentCode',
142             'name',
143             'transliteration',
144             'altNames'
145         ]));
146 
147         if (!$lang->save()) {
148             abort(500);
149         }
150 
151         // Update alphabets.
152         $alphabets = $this->getAlphabets($this->request->get('alphabets', ''));
153         $alphabetIDs = $alphabets->map(function($item) {
154             return $item->id;
155         })->toArray();
156 
157         $lang->alphabets()->sync($alphabetIDs);
158 
159         // Update countries.
160         $countries = $this->getCountries($this->request->get('countries', ''));
161         $countryIDs = $countries->map(function($item) {
162             return $item->id;
163         })->toArray();
164 
165         $lang->countries()->sync($countryIDs);
166 
167         // Send success message to client, and a thank you.
168         Session::push('messages', 'The details for <em>'. $lang->name .
169             '</em> were successfully saved, thanks :)');
170 
171         // Return URI
172         switch ($this->request->get('return'))
173         {
174             case 'admin':
175                 $return = route('admin.language.index');
176                 break;
177 
178             case 'edit':
179                 $return = $lang->editUri;
180                 break;
181 
182             case 'finish':
183             case 'summary':
184             default:
185                 $return = $lang->uri;
186         }

280     /**
281      * @param string $query
282      * @return mixed
283      */
284     public function search($query = '')
285     {
286         // This method should really only be called through the API.
287         if (Request::method() != 'POST' && env('APP_ENV') == 'production') {
288             abort(405);
289         }
290 
291         // Performance check
292         $query  = trim(preg_replace('/[\s+]/', ' ', strip_tags((string) $query)));
293         if (strlen($query) < 2) {
294             return $this->abort(400, 'Query too short');
295         }
296 
297         $offset = min(0, (int) Request::get('offset', 0));
298         $limit = min(1, max(100, (int) Request::get('limit', 100)));
299 
300         $langs = Language::search($query, $offset, $limit);
301 
302         // Format results
303         $results  = [];
304         $semantic = (bool) Request::has('semantic');
305         if (count($langs)) {
306             foreach ($langs as $lang) {
307                 $results[] = $semantic ? array_add($lang->toArray(), 'title', $lang->name) : $lang->toArray();
308             }
309         }
310 
311         // return $this->send(['query' => $query, 'results' => $results]);
312         return $this->send($semantic ? $results : compact('query', 'results'));
313     }

316      * Shortcut to retrieve a language object.
317      *
318      * @param string $id    Either the ISO 639-3 language code or language ID.
319      * @return \App\Models\Language|null
320      */
321     private function getLanguage($id, array $embed = ['parent'])
322     {
323         // Performance check.
324         if (empty($id) || is_numeric($id) || !is_string($id)) {
325             return null;
326         }
327 
328         // Find langauge by code.
329         if (strlen($id) == 3 || strlen($id) == 7) {
330             $lang = Language::findByCode($id, $embed);
331         }
332 
333         // Find language by ID.
334         else {
335             $lang = Language::with($embed)->find($id);
336         }
337 
338         return $lang;
339     }

341     /**
342      * Retrieves alphabet models.
343      *
344      * @param string $raw   Alphabet codes, separated by commas.
345      * @return Illuminate\Support\Collection
346      */
347     protected function getAlphabets($raw = null)
348     {
349         $alphabets = new Collection;
350 
351         // Retrieve alphabet models.
352         $raw = trim($raw);
353         if (strlen($raw))
354         {
355             foreach (@explode(',', $raw) as $code)
356             {
357                 if ($alphabet = Alphabet::findByCode($code)) {
358                     $alphabets->push($alphabet);
359                 }
360             }
361         }
362 
363         return $alphabets;
364     }
