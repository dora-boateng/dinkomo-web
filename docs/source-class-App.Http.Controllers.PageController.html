 20     /**
 21      * About the app.
 22      *
 23      * @param string $topic
 24      * @return View
 25      */
 26     public function about($topic = '')
 27     {
 28         $data = [];
 29 
 30         switch ($topic)
 31         {
 32             case 'api':
 33             case 'author':
 34             case 'sponsors':
 35             case 'translation-engine':
 36                 $view = 'pages.about.'. $topic;
 37                 break;
 38 
 39             case 'learning-app':
 40             case 'stats':
 41                 $view = 'pages.about.'. $topic;
 42 
 43                 // Retrieve top languages.
 44                 $topLangs = [];
 45                 $topLangQuery = DB::table('definitions')
 46                                 ->selectRaw('main_language_code AS code')
 47                                 ->selectRaw('COUNT(main_language_code) AS total')
 48                                 ->whereRaw('LENGTH(main_language_code) > 2')
 49                                 ->groupBy('code')
 50                                 ->orderBy('total', 'desc')
 51                                 ->get();
 52 
 53                 for ($i = 0; $i < min(5, count($topLangQuery)); $i++)
 54                 {
 55                     $lang = Language::findByCode($topLangQuery[$i]->code);
 56                     $topLangs[] = [
 57                         'name' => $lang->name,
 58                         'code' => $lang->code,
 59                         'total' => $topLangQuery[$i]->total,
 60                     ];
 61                 }
 62 
 63                 $data = [
 64                     'topLanguages' => $topLangs
 65                 ];
 66                 break;
 67 
 68             case 'story':
 69                 $view = 'pages.about.story';
 70                 $data = [
 71                     'defs' => Definition::count(),
 72                     'langs' => Language::count()
 73                 ];
 74                 break;
 75 
 76             case 'team':
 77                 $view = 'pages.about.team.index';
 78                 break;
 79 
 80             default:
 81                 $view = 'pages.about.index';
 82                 $data = [
 83                     'defs' => Definition::count(),
 84                     'langs' => Language::count()
 85                 ];
 86         }
 87 
 88         return view($view, $data);
 89     }
