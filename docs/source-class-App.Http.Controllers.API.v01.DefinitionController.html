204 
205         // Add definition to database.
206         $definition->fill($data);
207         if (!$definition->save()) {
208             return response('Could Not Save Definition.', 500);

210 
211         // Add language relations.
212         $languageCodes = Request::input('languages');
213         if (is_array($languageCodes))
214         {
215             $languageIDs = [];
