141     /**
142      * Looks up an alphabet model by code.
143      *
144      * @param string|\App\Models\Alphabet $code
145      * @return \App\Models\Alphabet|null
146      */
147     public static function findByCode($code)
148     {
149         // Performance check.
150         if ($code instanceof static) {
151             return $code;
152         }
153 
154         // Retrieve alphabet by code.
155         $code = preg_replace('/[^a-z\-]/', '', strtolower($code));
156 
157         return $code ? static::where('code', '=', ucfirst($code))->first() : null;
158     }

166     /**
167      * @param string $term      Search query.
168      * @param array $options    Search options.
169      * @return Builder
170      */
171     protected static function getSearchQueryBuilder($term, array $options = [])
172     {
173         $builder = DB::table('alphabets AS a')
174 
175             // TODO: Search language names and codes for languages that use "this" alphabet.
176             // ...
177 
178             // Create temporary score columns so we can sort the IDs.
179             ->selectRaw(
180                 'a.id, '.
181                 'a.name = ? AS name_score, '.
182                 'a.name LIKE ? AS name_score_low, '.
183                 'MATCH(a.transliteration) AGAINST(?) AS transliteration_score, '.
184                 'a.code = ? AS code_score, '.
185                 'a.script_code = ? AS script_code_score, '.
186                 'a.letters LIKE ? AS letters_score ',
187                 [$term, '%'.$term.'%', $term, $term, $term, '%'.$term.'%']
188             )
189 
190             // Try to search in a relevant way.
191             ->whereRaw(
192                 '('.
193                     'a.name = ? OR '.
194                     'a.name LIKE ? OR '.
195                     'MATCH(a.transliteration) AGAINST(?) OR '.
196                     'a.code = ? OR '.
197                     'a.script_code = ? OR '.
198                     'a.letters LIKE ? '.
199                 ')',
200                 [$term, '%'.$term.'%', $term, $term, $term, '%'.$term.'%']
201             );
202 
203         // Limit scope to a specific language.
204         if (isset($options['lang']) && $lang = Language::findByCode($options['lang'])) {
205             $builder->join('alphabet_language AS pivot', 'pivot.alphabet_id', '=', 'a.id')
206                 ->where('pivot.language_id', '=', DB::raw($lang->id));
207         }
208 
209         return $builder;
210     }

239     /**
240      * Normalizes the search score and formats a model for search results.
241      *
242      * @param object $alphabet
243      * @param object $scores
244      * @param float $maxScore
245      */
246     protected static function normalizeSearchResult($alphabet, $scores, $maxScore)
247     {
248         // If the name is an exact match, assign max score.
249         if ($scores->name_score > 0) {
250             $alphabet->score = 1;
251         }
252 
253         // If the code is an exact match, assign max score.
254         elseif ($scores->code_score > 0) {
255             $alphabet->score = 1;
256         }
257 
258         // In any other case, assign a score out of 0.9.
259         else {
260             $alphabet->score = $scores->total * 0.9 / $maxScore;
261         }
262     }
