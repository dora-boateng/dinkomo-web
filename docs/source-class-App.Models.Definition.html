 508     /**
 509      * @param string $term      Search query.
 510      * @param array $options    Search options.
 511      * @return Builder
 512      */
 513     protected static function getSearchQueryBuilder($term, array $options = [])
 514     {
 515         $builder = DB::table('definitions AS d')
 516 
 517             // Join the titles table so we can search its columns.
 518             ->leftJoin('definition_titles AS i', 'i.definition_id', '=', 'd.id')
 519 
 520             // Join the translations table so we can search its columns.
 521             ->leftJoin('translations AS t', 't.definition_id', '=', 'd.id')
 522 
 523             // Create temporary score columns so we can sort the IDs.
 524             ->selectRaw(
 525                 'd.id, '.
 526                 'i.title = ? AS title_score, '.
 527                 'i.title LIKE ? AS title_score_low, '.
 528                 'MATCH(i.transliteration) AGAINST(?) AS transliteration_score, '.
 529                 'MATCH(t.practical) AGAINST(?) AS practical_score, '.
 530                 't.practical LIKE ? AS practical_score_low, '.
 531                 't.practical = ? AS practical_score_multiplier, '.
 532                 'MATCH(t.literal) AGAINST(?) AS literal_score, '.
 533                 'MATCH(t.meaning) AGAINST(?) AS meaning_score ',
 534                 [$term, '%'.$term.'%', $term, $term, '%'.$term.'%', $term, $term, $term]
 535             )
 536 
 537             // Try to search in a relevant way.
 538             ->whereRaw(
 539                 '('.
 540                     'i.title = ? OR '.
 541                     'i.title LIKE ? OR '.
 542                     'MATCH(i.transliteration) AGAINST(?) OR '.
 543                     'MATCH(t.practical) AGAINST(?) OR '.
 544                     't.practical LIKE ? OR '.
 545                     'MATCH(t.literal) AGAINST(?) OR '.
 546                     'MATCH(t.meaning) AGAINST(?)'.
 547                 ')',
 548                 [$term, '%'.$term.'%', $term, $term, '%'.$term.'%', $term, $term]
 549             );
 550 
 551         // Limit scope to a specific language.
 552         if (isset($options['lang']) && $lang = Language::findByCode($options['lang'])) {
 553             $builder->join('definition_language AS pivot', 'pivot.definition_id', '=', 'd.id')
 554                 ->where('pivot.language_id', '=', DB::raw($lang->id));
 555         }
 556 
 557         // Limit scope to a specific definition type.
 558         if (isset($options['type']) && in_array($options['type'], static::types())) {
 559             $builder->where('d.type', '=', DB::raw(static::getTypeConstant($options['type'])));
 560         }
 561 
 562         // Limit scope to certain tags
 563         if (isset($options['tags']) && count($options['tags'])) {
 564             $tagIds = Tag::whereIn('title', $options['tags'])->lists('id');
 565 
 566             $builder->join('definition_tag AS tag_pivot', 'tag_pivot.definition_id', '=', 'd.id')
 567                 ->whereIn('tag_pivot.tag_id', $tagIds);
 568         }
 569 
 570         return $builder;
 571     }

 587             $rawScore->literal_score * 0.8 +

 603     /**
 604      * Normalizes the search score and formats a model for search results.
 605      *
 606      * @param object $definition
 607      * @param object $scores
 608      * @param float $maxScore
 609      */
 610     protected static function normalizeSearchResult($definition, $scores, $maxScore)
 611     {
 612         // If a title is an exact match, assign max score.
 613         if ($scores->title_score > 0) {
 614             $definition->score = 1;
 615         }
 616 
 617         // If a translation is an exact match, assign second-highest score.
 618         elseif ($scores->practical_score_multiplier > 0) {
 619             $definition->score = 0.95;
 620         }
 621 
 622         // In any other case, assign a score out of 0.9.
 623         else {
 624             $definition->score = $scores->total * 0.9 / $maxScore;
 625         }
 626 
 627         // Embeds.
 628         $definition->setAttribute('uri', $definition->uri);
 629         $definition->setAttribute('mainTitle', $definition->mainTitle);
 630         $definition->setAttribute('translation', $definition->translation);
 631         $definition->setAttribute('mainLanguage', $definition->mainLanguage);
 632         $definition->mainLanguage->setAttribute('uri', $definition->mainLanguage->uri);
 633     }
