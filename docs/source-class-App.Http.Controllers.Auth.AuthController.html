107     /**
108      * Obtain the user information from the OAuth provider.
109      *
110      * @param string $provider
111      * @return Response
112      */
113     public function handleProviderCallback($provider)
114     {
115         // Performance check.
116         if (!in_array($provider, $this->oAuthProviders)) {
117             abort(400);
118         }
119 
120         $user = Socialite::driver($provider)->user();
121 
122         // OAuth Two Providers
123         $token = $user->token;
124         $refreshToken = $user->refreshToken; // not always provided
125         $expiresIn = $user->expiresIn;
126 
127         // OAuth One Providers
128         $token = $user->token;
129         $tokenSecret = $user->tokenSecret;
130 
131         // All Providers
132         $user->getId();
133         $user->getNickname();
134         $user->getName();
