 62     /**
 63      * Uploads a backup file.
 64      *
 65      * @param   Illuminate\Http\UploadedFile    $file
 66      * @return  App\Factories\BackupFactory
 67      *
 68      * @throws Exception
 69      */
 70     public function upload(File $file)
 71     {
 72         // Retrieve filename.
 73         $file = $this->request->file('file');
 74         $filename = $file->getClientOriginalName();
 75 
 76         // Make sure a file with the same name doesn't already exist.
 77         if ($this->storage->exists($filename)) {
 78             throw new Exception('Backup file already exists.');
 79         }
 80 
 81         // Upload file to backups disk.
 82         $handle = fopen($file->getRealPath(), 'r');
 83         if (! $this->storage->put($filename, $handle)) {
 84             fclose($handle);
 85             throw new Exception('Could not upload backup file.');
 86         }
 87 
 88         fclose($handle);
 89         $this->setMessage('Backup file successfully uploaded.');
 90 
 91         return $this;
 92     }

 94     /**
 95      * Restores a backup file.
 96      *
 97      * @param   string  $filename
 98      * @param   array   $options
 99      * @return  void
100      *
101      * @throws  Exception
102      */
103     public function restore($filename, array $options = [])
104     {
105         // Performance check.
106         $file = $this->getPath($filename);
107         if (! $this->storage->exists($file)) {
108             throw new Exception('Can\'t find backup file "'.$file.'"');
109         }
110 
111         // Reset start time.
112         $this->startTime = time();
113 
114         // Run some pre-restore checks & tasks.
115         $this->msg('Putting app in maintenance mode...');
116         Artisan::call('down');
117 
118         // Unpack backup file.
119         $restoreId = 'restore-'.date('Ymd').'-'.substr(md5(microtime()), 20);
120         if (! $this->tempStorage->makeDirectory($restoreId)) {
121             throw new Exception('Could not create temp directory to unpack backup file.');
122         }
123 
124         if (! $this->tempStorage->put($restoreId.'/data.tar.gz', $this->storage->get($file))) {
125             $this->tempStorage->deleteDirectory($restoreId);
126 
127             throw new Exception('Could not copy backup file to temp directory.');
128         }
129 
130         // Extract backup file.
131         $phar = new PharData($this->getDirName($restoreId).'/data.tar.gz');
132         try {
133             $phar->decompress();
134             $phar->extractTo($this->getDirName($restoreId));
135         } catch (Exception $e) {
136             unset($phar);
137             Phar::unlinkArchive($this->getDirName($restoreId).'/data.tar.gz');
138             $this->tempStorage->deleteDirectory($restoreId);
139 
140             throw new Exception($e->getMessage());
141         }
142 
143         // Retrieve meta data.
144         $meta = $phar->getMetaData();
145         if (empty($meta)) {
146             if (! $this->tempStorage->exists($restoreId.'/meta.yaml')) {
147                 unset($phar);
148                 Phar::unlinkArchive($this->getDirName($restoreId).'/data.tar');
149                 Phar::unlinkArchive($this->getDirName($restoreId).'/data.tar.gz');
150                 $this->tempStorage->deleteDirectory($restoreId);
151 
152                 throw new Exception('Could not find metadata for backup file.');
153             }
154 
155             $meta = Yaml::parse($this->tempStorage->get($restoreId.'/meta.yaml'));
156         }
157 
158         // Remove phar files.
159         unset($phar);
160         Phar::unlinkArchive($this->getDirName($restoreId).'/data.tar');
161         Phar::unlinkArchive($this->getDirName($restoreId).'/data.tar.gz');
162 
163         // Refresh database.
164         if (isset($options['refresh-db'])) {
165             $this->msg('Refreshing migrations...');
166             Artisan::call('migrate:refresh');
167         }
168 
169 
170         // Restore backup.
171         $this->importHelper->setDataMeta($meta);
172         foreach ($this->resourceLimits as $resource => $limit) {
173             // Performance check.
174             if (! isset($meta[$resource]) || $meta[$resource]['files'] < 1) {
175                 continue;
176             }
177 
178             $this->msg("Loading {$meta[$resource]['files']} {$resource} files...");
179 
180             // Setup our DataImportFactory.
181             $this->importHelper->setDataModel('App\\Models\\'.ucfirst($resource));
182 
183 
184             // Loop through each file and import data using the DataImportFactory.
185             for ($i = 0; $i < $meta[$resource]['files']; $i++) {
186                 $dataFile = "{$restoreId}/{$resource}-{$i}.{$meta['format']}";
187                 if (! $this->tempStorage->exists($dataFile)) {
188                     continue;
189                 }
190 
191                 // Retrieve raw data.
192                 $data = $this->tempStorage->get($dataFile);
193                 $data = $meta['format'] == 'yaml' ? Yaml::parse($data) : json_decode($data, true);
194 
195                 // File checksum.
196                 if (! isset($options['skipChecksum']) || ! $options['skipChecksum']) {
197                     if ($meta[$resource]['checksums'][$i] !== $this->checksum($data, $meta['checksum-method'])) {
198                         $this->tempStorage->deleteDirectory($restoreId);
199 
200                         throw new Exception(
201                             "Checksum failed for {$resource}-{$i}.{$meta['format']} (expected \"".
202                             $meta[$resource]['checksums'][$i].'", got "'.
203                             $this->checksum($data, $meta['checksum-method']).'")'
204                         );
205                     }
206                 }
207 
208                 // Import data.
209                 try {
210                     $this->importHelper->setDataArray($data);
211                     $results = $this->importHelper->importDataSet();
212 
213                     foreach ($results->getMessages() as $msg) {
214                         $this->msg("File #$i: {$msg}");
215                     }
216                 } catch (Exception $e) {
217                     $this->tempStorage->deleteDirectory($restoreId);
218 
219                     throw new Exception($e->getMessage().' in '.$e->getFile().' on line '.$e->getLine());
220                 }
221             }
222         }
223 
224         // Delete temporary folder.
225         $this->tempStorage->deleteDirectory($restoreId);
226 
227         // Last checks & tasks.
228         Artisan::call('up');
229 
230         return $this;
231     }

261     /**
262      * Retrieves the relative path of a backup file.
263      *
264      * @param   string  $filename
265      * @param   int     $timestamp
266      *
267      * @throws  Exception
268      */
269     public function getPath($filename, $timestamp = null)
270     {
271         $fullName = null;
272 
273         // If a timestamp was provided, look for that specific backup.
274         if ($timestamp > 0) {
275             $fullName = date('Y-m', $timestamp).'/'.$filename;
276 
277             if (! $this->storage->exists($fullName)) {
278                 throw new Exception('Timestamp-filename combination does not exist.');
279             }
280         }
281 
282         // If none was provided, look for the latest backup matching the filename.
283         else {
284             $files = $this->storage->allFiles('/');
285 
286             foreach ($files as $file) {
287                 if (strpos($file, $filename) !== false) {
288                     $fullName = $file;
289                 }
290             }
291         }
292 
293         // Make sure a file was found.
294         if (! $fullName) {
295             throw new Exception('Backup file not found.');
296         }
297 
298         return $fullName;
299     }

311     /**
312      * @param   string  $msg

314     protected function msg($msg = '')
315     {
316         // Prepend elapsed time.
317         $time = '';
318         $sec = time() - $this->startTime;
319         if ($sec < 60) {
320             $time = $sec.' sec';
321         } else {
322             $mins = floor($sec / 60);
323             $time = $mins.' mins '.($sec - $mins * 60).' sec';
324         }
325         echo "[$time]";
326 
327         // Print message.
328         if (strlen($msg)) {
329             echo " $msg";
330         }
331 
332         // Newline character.
333         echo "\n";
334     }
